-- menu 定义

CREATE TABLE `menu` (
  `menu_name` varchar(100) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `menu_url` varchar(100) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `create_time` datetime DEFAULT NULL,
  `id` int NOT NULL AUTO_INCREMENT,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=5 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='菜单';


-- `user` 定义

CREATE TABLE `user` (
  `user_id` int NOT NULL AUTO_INCREMENT,
  `user_name` varchar(100) NOT NULL,
  `create_time` timestamp NULL DEFAULT NULL,
  `update_time` timestamp NULL DEFAULT NULL,
  `status` int DEFAULT NULL COMMENT '用户状态',
  `password` varchar(100) NOT NULL COMMENT '密码',
  `email` varchar(100) DEFAULT NULL,
  `phone` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`user_id`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;


-- warehouse 定义

CREATE TABLE `warehouse` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `goods_name` varchar(100) NOT NULL COMMENT '商品名称',
  `goods_code` varchar(50) NOT NULL COMMENT '商品编码（唯一）',
  `spec` varchar(100) DEFAULT NULL COMMENT '规格',
  `quantity` int NOT NULL COMMENT '入库数量',
  `in_date` date NOT NULL COMMENT '入库日期',
  `remark` varchar(500) DEFAULT NULL COMMENT '备注',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_goods_code` (`goods_code`) COMMENT '商品编码唯一'
) ENGINE=InnoDB AUTO_INCREMENT=17 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='商品入库表';


-- warehouse_in 定义

CREATE TABLE `warehouse_in` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `goods_code` varchar(50) NOT NULL COMMENT '商品编码（唯一）',
  `spec` varchar(100) DEFAULT NULL COMMENT '规格',
  `quantity` int NOT NULL COMMENT '入库数量',
  `in_date` date NOT NULL COMMENT '入库日期',
  `remark` varchar(500) DEFAULT NULL COMMENT '备注',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_goods_code` (`goods_code`) COMMENT '商品编码唯一'
) ENGINE=InnoDB AUTO_INCREMENT=16 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='商品入库表';


-- warehouse_out 定义

CREATE TABLE `warehouse_out` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `goods_code` varchar(64) NOT NULL COMMENT '商品编码',
  `out_quantity` int NOT NULL COMMENT '出库数量',
  `out_date` date NOT NULL COMMENT '出库日期',
  `remark` varchar(512) DEFAULT NULL COMMENT '备注',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_goods_code` (`goods_code`)
) ENGINE=InnoDB AUTO_INCREMENT=4 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='出库记录表';
